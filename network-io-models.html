三、内核等待队列机制 — 所有模型的基石

/*
 * 无论 Blocking、select、epoll 还是 io_uring，
 * 底层都依赖 wait_queue + schedule() 机制：
 * 把进程/线程从 runqueue 摘下来，挂到等待队列上，
 * 等事件到了再放回去。
 */

// ===== Blocking: 进程挂在 socket 的等待队列上 =====
int inet_wait_for_connect(struct sock *sk, ...) {
    DEFINE_WAIT_FUNC(wait, woken_wake_function);
    add_wait_queue(sk_sleep(sk), &wait);   // 挂到 sk->sk_wq
    while (sk->sk_state != TCP_ESTABLISHED) {
        wait_woken(&wait, TASK_INTERRUPTIBLE, timeo); // schedule()
    }
    remove_wait_queue(sk_sleep(sk), &wait);
}

// ===== select: 进程挂在每个 fd 的等待队列上 =====
int do_select(int n, fd_set_bits *fds, ...) {
    for (i = 0; i < n; i++) {                // O(n) 遍历
        poll_initwait(&table);
        if (FD_ISSET(i, fds))
            fd[i]->f_op->poll(fd[i], &table);  // 注册到每个 fd
    }
    poll_schedule_timeout(&table, ...);        // schedule()
}

// ===== epoll: 进程挂在 epoll 自己的等待队列上 =====
int ep_poll(struct eventpoll *ep, ...) {
    __add_wait_queue_exclusive(&ep->wq, &wait); // 挂到 ep->wq
    for (;;) {
        if (!list_empty(&ep->rdllist)) break;   // 就绪链表非空
        schedule_hrtimeout_range(...);            // schedule()
    }
}

// fd 就绪时回调 (软中断上下文):
int ep_poll_callback(struct eventpoll *ep, ...) {
    list_add_tail(&epi->rdllink, &ep->rdllist); // 加入就绪链表 O(1)
    wake_up(&ep->wq);                          // 唤醒 epoll 上的进程
}

// ===== io_uring: 内核工作线程 + CQ ring =====
// 提交: SQ ring → io_submit_sqes() → 内核异步执行
// 完成: 内核写 CQ ring → 用户态直接读取 (mmap, 无 syscall)
// 等待: io_cqring_wait() → schedule() 或 直接 poll CQ ring
